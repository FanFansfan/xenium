141  template <bool Weak, class... Args>
142  bool do_try_push(Args&&... args) {
143  cell* c;
144  std::size_t pos = enqueue_pos.load(std::memory_order_relaxed);
145  for (;;) {
146  c = &cells[pos & index_mask];
147  // (3) - this acquire-load synchronizes-with the release-store (2)
148  std::size_t seq = c->sequence.load(std::memory_order_acquire);
149  if (seq == pos) {
150  if (enqueue_pos.compare_exchange_weak(pos, pos + 1, std::memory_order_relaxed))
151  break;
152  } else {
153  if (Weak) {
154  if (seq < pos)
155  return false;
156  else
157  pos = enqueue_pos.load(std::memory_order_relaxed);
158  } else {
159  auto pos2 = enqueue_pos.load(std::memory_order_relaxed);
160  if (pos2 == pos && dequeue_pos.load(std::memory_order_relaxed) + index_mask + 1 == pos)
161  return false;
162  pos = pos2;
163  }
164  }
165  }
166  assign_value(c->value, std::forward<Args>(args)...);
167  // (4) - this release-store synchronizes-with the acquire-load (1)
168  c->sequence.store(pos + 1, std::memory_order_release);

173  bool do_try_pop(T& result) {
174  cell* c;
175  std::size_t pos = dequeue_pos.load(std::memory_order_relaxed);
176  for (;;) {
177  c = &cells[pos & index_mask];
178  // (1) - this acquire-load synchronizes-with the release-store (4)
179  std::size_t seq = c->sequence.load(std::memory_order_acquire);
180  auto new_pos = pos + 1;
181  if (seq == new_pos) {
182  if (dequeue_pos.compare_exchange_weak(pos, new_pos, std::memory_order_relaxed))
183  break;
184  } else {
185  if (Weak) {
186  if (seq < new_pos)
187  return false;
188  pos = dequeue_pos.load(std::memory_order_relaxed);
189  } else {
190  auto pos2 = dequeue_pos.load(std::memory_order_relaxed);
191  if (pos2 == pos && enqueue_pos.load(std::memory_order_relaxed) == pos)
192  return false;
193  pos = pos2;
194  }
195  }
196  }
197  result = std::move(c->value);
198  // (2) - this release-store synchronizes-with the acquire-load (3)
199  c->sequence.store(pos + index_mask + 1, std::memory_order_release);
200  return true;
